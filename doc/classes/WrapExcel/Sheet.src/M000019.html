    def initialize(win32_worksheet)
      @sheet = win32_worksheet
      if @sheet.ProtectContents
        @sheet.Unprotect
        @end_row = last_row
        @end_column = last_column
        @sheet.Protect
      else
        @end_row = last_row
        @end_column = last_column
      end
    end
